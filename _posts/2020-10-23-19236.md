// 19236 청소년 상어
/*
    1. 물고기 번호 저장 int mp[MAX][MAX];
    2. 번호 - 위치 방향 저장
    fish[번호] = {
        int X;
        int Y;
        int D;
    } 
*/
#include <bits/stdc++.h>
using namespace std;
#define MAX 21
#define INF 1999999999
int dx[8] = {0, -1, -1, -1, 0, 1, 1, 1};
int dy[8] = {-1, -1, 0, 1, 1, 1, 0, -1};
int mp[4][4];
int ans = 0;
struct fish
{
    int Y;
    int X;
    int D;
    bool isDead = false;
};
fish F[26];
void seeM()
{
    cout << "\n";
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
        {
            cout << mp[i][j] << " ";
        }
        cout << "\n";
    }
    cout << "\n";
}
void fishMove()
{
    int rot = 0;
    for (int i = 1; i <= 16; i++)
    {
        seeM();
        fish cur = F[i];
        if (cur.isDead)
            continue;
        int d = cur.D;
        int nx = cur.X + dx[d];
        int ny = cur.Y + dy[d];
        //출구가 없을경우 가만히 있는다.
        if (rot >= 8)
        {
            rot = 0;
            continue;
        }
        //막히면 다시
        if (ny < 0 || ny >= 4 || nx < 0 || nx >= 4 || mp[ny][nx] == -1)
        {
            rot++;
            F[i].D = (F[i].D + 1) % 8;
            i = i - 1;
            continue;
        }
        rot = 0;
        //물고기가 있으면 rotation
        if (mp[ny][nx])
        {
            //움직이는 자리에 있는 물고기
            int nxNum = mp[ny][nx];
            //mp위치 바꾸고
            mp[ny][nx] = i;
            mp[cur.Y][cur.X] = nxNum;
            //물고기 위치 정보 바꾸고
            fish tmpF;
            tmpF = F[nxNum];
            F[nxNum].X = F[i].X;
            F[nxNum].Y = F[i].Y;
            F[i].X = tmpF.X;
            F[i].Y = tmpF.Y;
            continue;
        }
        //물고기 없으면 그냥 이동
        mp[cur.Y][cur.X] = 0;
        mp[ny][nx] = i;
        F[i].Y = ny;
        F[i].X = nx;
    }
}

void eatFish(fish shark,int sum)
{
    int X = shark.X;
    int Y = shark.Y;
    int D = shark.D;
    int fMove =0;
    //bfs shark , sum , N(횟수)
    queue<pair<fish,pair<int,int>>> q;
    q.push({shark,{0,0}});
    while(!q.empty())
    {
        fish cur = q.front().first;
        int curS = q.front().second.first;
        int N = q.front().second.second;
        mp[cur.Y][cur.X] = -1;
        if(fMove != N)
        {
            fishMove();
            fMove++;
        }

    }
    for(int i=0;i<4;i++)
    {
        int nx = X+dx[D];
        int ny = Y+dy[D];
        if(mp[ny][nx] == 0) continue;
        if(nx <0 || nx >=4 || ny <0 || ny>=4) continue;

    }
}

{
    //ios::sync_with_stdio(false);

    {
        for (int j = 0; j < 4; j++)

            b -= 1;
            mp[i][j] = a;
            F[a] = {i, j, b};
        }
    }

    fishMove();
